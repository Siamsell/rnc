from contextlib import contextmanager

class DatabasePool:
    def __init__(self):
        self._pool = None
    
    def create_pool(self, **kwargs):
        """Initialise le pool avec les paramÃ¨tres de connexion"""
        self._pool = pool.ThreadedConnectionPool(
            minconn=1,
            maxconn=20,
            **kwargs
        )
        logging.info("Pool de connexions PostgreSQL initialisÃ©")

    @contextmanager
    def get_connection(self):
        """Gestion contextuelle des connexions"""
        conn = self._pool.getconn()
        try:
            yield conn
        except Exception as e:
            logging.error(f"Erreur de connexion: {e}")
            conn.rollback()
            raise
        finally:
            self._pool.putconn(conn)

    def close_all(self):
        """Ferme toutes les connexions du pool"""
        self._pool.closeall()
        logging.info("Toutes les connexions du pool ont Ã©tÃ© fermÃ©es")
